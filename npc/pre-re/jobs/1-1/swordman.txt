function	script	F_JobSwdStaff	{
	mes "[Funcionário do Salão do Teste]";
	mes "Você desiste?";
	next;
	if (select("Sim:Não") == 1) {
		mapannounce "job_sword1",strcharinfo(0) + " desistiu do teste.",bc_map;
		warp "izlude_in",65,165;
		end;
	}
	mes "[Funcionário do Salão do Teste]";
	mes "Bravo! Não desista!";
	close;
}
